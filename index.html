// Función para enviar datos al servidor async function enviarCorreo() { const datos = { from_name: "Alfonso", to_name: "Receptor", subject: "Mensaje de Prueba", message: "Hola, este es un mensaje desde el experimento de PsychoPy." }; try { const respuesta = await fetch('http://localhost:3000/send-email', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(datos) }); if (respuesta.ok) { const resultado = await respuesta.json(); console.log('Respuesta del servidor:', resultado); alert('Correo enviado exitosamente.'); } else { console.error('Error al enviar correo:', respuesta.status); alert('Error al enviar el correo.'); } } catch (error) { console.error('Error:', error); alert('Ocurrió un error al conectar con el servidor.'); } } // Llamada de prueba enviarCorreo();
